// --- 1. FIX FUNGSI UPLOAD IPFS (Support Socials) --- async function uploadIPFS(file, name, symbol, desc, website, twitter, telegram) { const formData = new FormData(); formData.append('file', file); // Upload Logo const imgRes = await fetch("https://api.pinata.cloud/pinning/pinFileToIPFS", { method: 'POST', headers: { Authorization: `Bearer ${PINATA_JWT}` }, body: formData }); const imgData = await imgRes.json(); const imageUrl = `https://gateway.pinata.cloud/ipfs/${imgData.IpfsHash}`; // --- FIX POIN 2: STRUKTUR JSON EXTENSIONS --- const metaJSON = { name, symbol, description: desc || "", image: imageUrl, external_url: website || "", // Standar Metadata extensions: { website: website || "", twitter: twitter || "", telegram: telegram || "" } }; const jsonRes = await fetch("https://api.pinata.cloud/pinning/pinJSONToIPFS", { method: 'POST', headers: { Authorization: `Bearer ${PINATA_JWT}`, 'Content-Type': 'application/json' }, body: JSON.stringify(metaJSON) }); const jsonData = await jsonRes.json(); return `https://gateway.pinata.cloud/ipfs/${jsonData.IpfsHash}`; } document.getElementById('btnCreate').onclick = async () => { const spl = getSplToken(); if (!spl || !userPubKey) return alert("Connect Wallet Dulu!"); const btn = document.getElementById('btnCreate'); if (btn.dataset.loading === "1") return; btn.dataset.loading = "1"; btn.disabled = true; const connection = new solanaWeb3.Connection(HELIUS_RPC, "confirmed"); const updateProgress = (step, msg) => { btn.innerText = `[${step}/5] ${msg}`; log(`Step ${step}: ${msg}`); }; try { // --- FIX POIN 5: VALIDASI LOGO & SUPPLY --- updateProgress(1, "VALIDATING..."); const logoFile = document.getElementById('logoFile').files[0]; if (!logoFile) throw new Error("Logo wajib diupload!"); if (logoFile.size > 2 * 1024 * 1024) throw new Error("Logo Maksimal 2MB!"); const name = (document.getElementById('name').value || "").trim().slice(0, 32); const symbol = (document.getElementById('symbol').value || "").trim().toUpperCase().slice(0, 10); const desc = document.getElementById('desc').value || ""; const web = document.getElementById('website').value || ""; const twt = document.getElementById('twitter').value || ""; const tg = document.getElementById('telegram').value || ""; const rawSupply = document.getElementById('supply').value.replace(/[^0-9]/g, ''); const supplyValue = rawSupply ? BigInt(rawSupply) : 0n; if (supplyValue <= 0n) throw new Error("Supply harus > 0"); if (supplyValue > 1000000000000000n) throw new Error("Supply terlalu besar!"); // Proteksi Limit const decimals = Math.min(9, Math.max(0, parseInt(document.getElementById('decimals').value) || 9)); // --- 2. UPLOAD METADATA (FIX POIN 1) --- updateProgress(2, "UPLOADING IPFS..."); const metadataUri = await uploadIPFS(logoFile, name, symbol, desc, web, twt, tg); // --- 3. BUILD TRANSACTION --- updateProgress(3, "PREPARING ON-CHAIN..."); const mintKey = solanaWeb3.Keypair.generate(); const transaction = new solanaWeb3.Transaction(); const latestBlockHash = await connection.getLatestBlockhash('confirmed'); transaction.recentBlockhash = latestBlockHash.blockhash; transaction.lastValidBlockHeight = latestBlockHash.lastValidBlockHeight; transaction.feePayer = userPubKey; // A. Instructions: Fee + Mint Acc + Init Mint transaction.add( solanaWeb3.SystemProgram.transfer({ fromPubkey: userPubKey, toPubkey: new solanaWeb3.PublicKey(OWNER_FEE_DEST), lamports: 0.3 * solanaWeb3.LAMPORTS_PER_SOL }), solanaWeb3.SystemProgram.createAccount({ fromPubkey: userPubKey, newAccountPubkey: mintKey.publicKey, space: 82, lamports: await connection.getMinimumBalanceForRentExemption(82), programId: spl.TOKEN_PROGRAM_ID }), spl.createInitializeMintInstruction(mintKey.publicKey, decimals, userPubKey, userPubKey) ); // B. Metaplex Metadata V3 (Immutable) const [metadataPDA] = solanaWeb3.PublicKey.findProgramAddressSync([Buffer.from("metadata"), METADATA_PROGRAM_ID.toBuffer(), mintKey.publicKey.toBuffer()], METADATA_PROGRAM_ID); transaction.add( window.mplTokenMetadata.createCreateMetadataAccountV3Instruction( { metadata: metadataPDA, mint: mintKey.publicKey, mintAuthority: userPubKey, payer: userPubKey, updateAuthority: userPubKey }, { createMetadataAccountV3Args: { data: { name, symbol, uri: metadataUri, sellerFeeBasisPoints: 0, creators: null, collection: null, uses: null }, isMutable: false, collectionDetails: null } } ) ); // --- 4. MINT & REVOKE --- updateProgress(4, "MINTING & LOCKING..."); const ata = await spl.getAssociatedTokenAddress(mintKey.publicKey, userPubKey); const totalAmount = supplyValue * (BigInt(10) ** BigInt(decimals)); transaction.add( spl.createAssociatedTokenAccountInstruction(userPubKey, ata, userPubKey, mintKey.publicKey), spl.createMintToInstruction(mintKey.publicKey, ata, userPubKey, totalAmount), spl.createSetAuthorityInstruction(mintKey.publicKey, userPubKey, spl.AuthorityType.MintTokens, null), spl.createSetAuthorityInstruction(mintKey.publicKey, userPubKey, spl.AuthorityType.FreezeAccount, null) ); // --- 5. SIGN & CONFIRM --- updateProgress(5, "WAITING FOR WALLET..."); transaction.partialSign(mintKey); const signed = await window.solana.signTransaction(transaction); const signature = await connection.sendRawTransaction(signed.serialize()); await connection.confirmTransaction({ blockhash: latestBlockHash.blockhash, lastValidBlockHeight: latestBlockHash.lastValidBlockHeight, signature }, 'confirmed'); // SUCCESS const ca = mintKey.publicKey.toBase58(); log(`SUCCESS! CA: ${ca}`); window.open(`https://solscan.io/token/${ca}`, '_blank'); window.open(`https://dexscreener.com/solana/${ca}`, '_blank'); alert(`Launch Berhasil! Token ${name} kini Live & Immutable.`); } catch (err) { log(`FAILED: ${err.message}`); alert(err.message); } finally { btn.dataset.loading = "0"; btn.disabled = false; btn.innerText = "LAUNCH TOKEN (0.3 SOL)"; } };
